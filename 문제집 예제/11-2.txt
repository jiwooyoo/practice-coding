using namespace std;   //표준 입출력을 위해
deque<int> q;

int calc(int a, int b) {
	return max(a + b, a*b);
}

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	string s;
	cin >> s;

	for (int i = 0; i < s.size(); i++) {
		q.push_back(s[i] - '0');
	}
	
	int res=q.front();
	q.pop_front();
	while (!q.empty()) {
		int n = q.front();
		q.pop_front();

		res=calc(res, n);
	}

	cout << res << "\n";
}
