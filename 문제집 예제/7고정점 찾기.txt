using namespace std;   //표준 입출력을 위해
vector<int> v;

int bin(int start, int end) {
	if (start > end) return -1;

	int mid = (start + end) / 2;
	
	if (v[mid] == mid) return mid;
	else if (v[mid] < mid) return bin(mid + 1, end);
	else return bin(start, mid - 1);
}

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int n, m;
	cin >> n;
	
	for (int i = 0; i < n; i++) {
		int key;
		cin >> key;
		v.push_back(key);
	}

	//sort(v.begin(), v.end());

	cout << bin(0, n - 1) <<"\n";
}
