#define INF 987654321
#define MAX 300001

using namespace std;   //표준 입출력을 위해
vector<int> v;

bool bin(int start, int end, int num) {
	if (start > end) return false;
	int mid = (start + end) / 2;
	if (v[mid] == num) return true;
	else if (v[mid] < num)
		return bin(mid + 1, end, num);
	else return bin(start, mid - 1, num);
}
int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	

	int n, m;
	cin >> n;
	
	for (int i = 0; i < n; i++) {
		int key;
		cin >> key;
		v.push_back(key);
	}

	sort(v.begin(), v.end());

	cin >> m;
	for (int i = 0; i < m; i++) {
		int key;
		cin >> key;
		cout << key << "\n";
		if (bin(0, n - 1, key))
			cout << "yes\n";
		else cout << "no\n";
	}
}
