using namespace std;   //표준 입출력을 위해
vector<int> v;

int bin(int start, int end, int num) {
	if (start > end) return -1;

	int mid = (start + end) / 2;
	int re,high,low;
	
	if (v[mid] == num) {
		re = mid;
		low = min(re, bin(start, mid - 1, num));
		high = max(re, bin(mid + 1, end, num));
	}
	else if (v[mid] < num) {
		return bin(start, mid - 1, num);
	}
	else {
		return bin(mid + 1, end, num);
	}

	return high - low;
}

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int n, m;
	cin >> n >> m;
	
	for (int i = 0; i < n; i++) {
		int key;
		cin >> key;
		v.push_back(key);
	}

	//sort(v.begin(), v.end());

	cout << bin(0, n - 1, m) <<"\n";
}
