#include<iostream> 
#include<algorithm>
#include<string>
#include<queue>
#include<stack>
#include<vector>
#include<memory.h>
#define INF 796796
#define MAX 10002

using namespace std;   //표준 입출력을 위해
vector<int> v;

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int n;
	cin >> n;

	for (int i = 0; i < n; i++) {
		int x;
		cin >> x;
		v.push_back(x);
	}

	sort(v.begin(), v.end());

	int res = 0;
	int cnt = 0;

	for (int i = 0; i < n; i++) {
		cnt++;
		if (cnt >= v[i]) {
			res++;
			cnt = 0;
		}
	}

	cout << res << "\n";
}
