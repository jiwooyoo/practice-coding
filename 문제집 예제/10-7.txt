using namespace std;   //표준 입출력을 위해
vector<int> v;
int root[100];

int parent(int num) {
	if (root[num] != num) {
		parent(root[num]);
	}
	return root[num];
}

void union_p(int a, int b) {
	int aa = parent(a);
	int bb = parent(b);
	if (aa < bb)
		root[bb] = aa;
	else root[aa] = bb;
}

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int n, m;
	cin >> n >> m;

	for (int i = 0; i <= n; i++) {
		root[i] = i;
	}
	for (int i = 0; i < m; i++) {
		int x, a, b;
		cin >> x >> a >> b;
		if (x == 0) {
			union_p(a,b);
		}
		else {
			if (parent(a) == parent(b))
				cout << "YES\n";
			else cout << "NO\n";
		}
	}
}
