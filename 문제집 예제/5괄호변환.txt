#include<iostream> 
#include<algorithm>
#include<string>
#include<queue>
#include<stack>
#include<vector>
#include<memory.h>

using namespace std;   //표준 입출력을 위해

bool check(string s) {
    stack<char> st;
	int len = s.length();
	for (int i = 0; i < len; i++) {
		if (s[i] == '(') st.push(s[i]);
		else {
			if (st.empty()) return false;
			else {
				st.pop();
			}
		}
	}
	return true;
}

string opposite(string s) {
	
	for (int i = 0; i < s.size(); i++) {
		if(s[i]=='(') s[i]=')';
        else s[i]='(';
	}
	
	return s;
}

string correct(string s) {
	int cnt = 0;
	int index;
	if (s == "") return s;

	for (int i = 0; i < s.size(); i++) {
		if (s[i] == '(')
			cnt++;
		else cnt--;

		if (cnt == 0) {
			index = i;
			break;
		}
	}

	string u = s.substr(0, index+1);
	string v = s.substr(index+1);

	string fir="",sec="";
	if (check(u)) {
		fir += u;
		sec += correct(v);
	}
	else {
        fir += "(";
        fir += correct(v);
        fir += ")";
        
		u = u.substr(1,u.size()-2);

		sec += opposite(u);
	}
    
	return fir+sec;
}

string solution(string p) {
    string answer = "";
    
    if (check(p)) answer=p;
	else {
		answer=correct(p);
	}
    
    return answer;
}
