#include<iostream> 
#include<algorithm>
#include<string>
#include<queue>
#include<stack>
#include<vector>
#include<memory.h>
#define INF 796796
#define MAX 10002

using namespace std;   //표준 입출력을 위해
int dp[MAX] = { 0, };
int arr[102] = { 0, };

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	int n,m;
	cin >> n >> m;

	memset(dp, MAX, sizeof(dp));
	for (int i = 0; i < n; i++) {
		cin >> arr[i];
		dp[arr[i]] = 1;
	}

	for (int i = 1; i <= m; i++) {
		for (int j = 0; j < n; j++) {
			if (i > arr[j]) dp[i] = min(dp[i],dp[i - arr[j]]+1);
		}
	}

	if (dp[m] >= MAX) cout << "-1\n";
	else cout << dp[m] << "\n";
}
