#include<iostream> 
#include<algorithm>
#include<string>
#include<queue>
#include<stack>
#include<vector>
#include<memory.h>
#define INF 796796
#define MAX 10002

using namespace std;   //표준 입출력을 위해

int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	string s;
	cin >> s;
	int num=0;
	vector<char> ch;

	for (int i = 0; i < s.size(); i++) {
		if (s[i] - '0' > 9) {
			ch.push_back(s[i]);
		}
		else {
			num+=(s[i] - '0');
		}
	}

	sort(ch.begin(), ch.end());

	for (int i = 0; i < ch.size(); i++) {
		cout << ch[i];
	}
	
	cout<<num << "\n";
}
